```yaml
# 项目计划书数据模型
project_plan:
  # 基础信息
  basic_info:

    project_code: ""           # 项目编号，格式：JYIS-YYYY-XXX
    domain: "D03"              # 固定为D03
    project_type: ""           # 类型：new_feature|optimization|bug_fix
    
  # 项目目标
  goals:
    business: []               # 业务目标列表
    technical: []              # 技术目标列表
    kpi: []                    # 成功指标KPI列表
    
  # 项目范围
  scope:
    includes: []               # 包含功能列表
    excludes: []               # 不包含功能列表
    
  # 资源需求
  resources:
    agents: {}                 # 智能体人力需求
    compute: {}                # 计算资源需求
    external: []               # 外部依赖列表
    
  # 时间计划
  schedule:
    start_date: ""             # 开始日期 YYYY-MM-DD
    end_date: ""               # 预计完成日期
    milestones: []             # 关键里程碑
    
  # 风险评估
  risks:
    technical: []              # 技术风险
    resource: []               # 资源风险
    mitigation: []             # 应对措施
    
  # 预算
  budget:
    development: 0             # 开发成本
    operation: 0               # 运营成本
    total: 0                   # 总预算
```

## 六、API接口定义

```python
# 项目计划书API

# 创建项目计划书
POST /api/v1/projects/plans
Request Body: <ProjectPlan YAML>
Response: { "plan_id": "string", "status": "created" }

# 获取项目计划书
GET /api/v1/projects/plans/{plan_id}
Response: <ProjectPlan YAML>

# 更新项目计划书
PUT /api/v1/projects/plans/{plan_id}
Request Body: <ProjectPlan YAML>

# 提交审批
POST /api/v1/projects/plans/{plan_id}/submit
Response: { "status": "pending_approval" }

# 审批
POST /api/v1/projects/plans/{plan_id}/approve
Request Body: { "level": "L1|L2|L3", "approved": true, "comment": "string" }
```
